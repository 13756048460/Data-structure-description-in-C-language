**带头节点的单链表**

```c
#include<stdio.h>

#define MAXSIZE 20

typedef int Status;
typedef int ElemType;

typedef sturct LNode{//定义单链表结点类型      
    ElemType date ;//每个结点存放一个数据元素
    struct LNode *next;//指针指向下一个结点
}LNode, *LinkList;

//创建单链表
void InitList(LinkList &l)
{
    L = (LNode*)malloc(sizeof(LNode));//分配一个头节点
    if (L != NULL) {
        L->next = NULL;
    }
}
//创建单链表，返回创建好的单链表
LinkList InitList() {
    LinkList L = (LinkList)malloc(sizeof(LNode));
    if (L != NULL) {
        L->next = NULL;
    }
    return L;
}

//在第i个位置插入元素e
void ListInsert(LinkList L, int i, ElemType e) {
    if (i < 1) 
        return ;   // 位置i非法（i至少为1）
    LNode *p = L;              // p指向头节点
    int j = 0;                 // 当前p指向第0个节点（头节点）
    // 寻找第i-1个节点（循环结束条件：p为空或j≥i-1）
    while (p != NULL && j < i - 1) {
        p = p->next;
        j++;
    }
    if (p == NULL) return false; // i超过链表长度+1，插入失败
    LNode *s = (LNode*)malloc(sizeof(LNode));
    if (s == NULL) return false; // 内存分配失败
    s->data = e;
    s->next = p->next;
    p->next = s;      // 将新节点插入链表
}
//头插法 
void HeadInsert(LinkList L, ElemType e) {
    if (L == NULL) return false; // 确保头节点存在
    LNode *s = (LNode *)malloc(sizeof(LNode));
    if (s == NULL) return false; // 内存分配失败
    s->data = e;        // 存入数据
    s->next = L->next;  // 新节点指向原首节点
    L->next = s;        // 头节点指向新节点
}
//尾插法
void TailInsert(LinkList L, ElemType e) {
    if (L == NULL) 
        return ;
    LNode *p = L;
    // 找到最后一个节点（p最终指向尾节点）
    while (p->next != NULL) {
        p = p->next;
    }
    LNode *s = (LNode*)malloc(sizeof(LNode));
    if (s == NULL)
        return ; // 内存分配失败
    s->data = e;
    s->next = NULL;    // 尾节点的next必须为NULL
    p->next = s;       // 原尾节点指向新节点
}

// 删除第i个位置的节点（i从1开始），并返回被删除的元素值
void ListDelete(LinkList L, int i, ElemType *e) {
    if (L == NULL || i < 1)
        return;
    LNode *p = L;       // p指向头节点
    int j = 0;          // 当前p指向头节点（第0个节点）
    // 寻找第i-1个节点（循环终止条件：p为空或找到第i-1个节点）
    while (p->next != NULL && j < i - 1) {
        p = p->next;
        j++;
    }
    if (p->next == NULL)
        return ;  // i超过链表长度
    LNode *q = p->next; // q为待删除节点（第i个节点）
    *e = q->data;       // 保存被删除节点的值
    p->next = q->next;  // 跳过q节点
    free(q);            // 释放内存
}

// 头删：删除链表的第一个数据节点
void HeadDelete(LinkList L, ElemType *e) {
    if (L == NULL || L->next == NULL)
        return ; // 空链表
    LNode *q = L->next; // q指向首元节点
    *e = q->data;       // 保存被删除节点的值
    L->next = q->next;  // 头节点指向第二个节点
    free(q);            // 释放内存
}

// 尾删：删除链表的最后一个节点
void TailDelete(LinkList L, ElemType *e) {
    if (L == NULL || L->next == NULL)
        return; // 空链表
    LNode *p = L;       // p从头节点开始遍历
    // 找到倒数第二个节点（p->next->next为NULL时停止）
    while (p->next->next != NULL) {
        p = p->next;
    }
    LNode *q = p->next; // q为尾节点
    *e = q->data;       // 保存被删除节点的值
    p->next = NULL;     // 倒数第二个节点成为新尾节点
    free(q);            // 释放内存
}

//按位查找，返回第i个节点（位序从1开始）
LNode* GetElem(LinkList L, int i) {
    if (i < 1)
        return NULL; // 位序非法
    LNode *p = L->next; // 从第一个元素节点开始
    int j = 1;          // 当前p指向第1个节点
    while (p != NULL && j < i) {
        p = p->next;
        j++;
    }
    return p; // 若i超出链表长度，返回NULL
    
// 按值查找，返回第一个数据域为e的节点
LNode* LocateElem(LinkList L, ElemType e) {
    LNode *p = L->next; // 跳过头节点
    while (p != NULL) {
        if (p->data == e) {
            return p;
        }
        p = p->next;
    }
    return NULL; // 未找到返回NULL
}
    
// 求链表长度（不包含头节点）
int Length(LinkList L) {
    int len = 0;
    LNode *p = L->next; // 从头节点后的第一个节点开始
    while (p != NULL) {
        len++;
        p = p->next;
    }
    return len;
}
```
